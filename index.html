Custom Calendar Component

Date Picker (Date Only)

Select a date. The value will be in format: YYYY-MM-DD

Selected Value: None

DateTime Picker (24-hour format)

Select a date and time. The value will be in format: YYYY-MM-DD HH:mm

Selected Value: None

DateTime Picker (12-hour format)

Select a date and time in 12-hour format. The value will be in format: YYYY-MM-DD HH:mm

Selected Value: None 
// Date Only Calendar
const dateCalendar = new LumiCalendar({
    target: '#calendar-date',
    enableDateTime: false,
    onChange: (value) => {
        console.log('Selected date:', value);
        document.getElementById('date-value').textContent = value || 'None';
    }
});

// DateTime Calendar (24-hour format - default)
const datetimeCalendar = new LumiCalendar({
    target: '#calendar-datetime',
    enableDateTime: true,
    hourFormat: '24', // Optional, '24' is the default
    startDay: 1,
    dayNames: ['Нед', 'Пон', 'Втор', 'Сря', 'Чет', 'Пет', 'Съб'],
    monthsNames: ['Януари', 'Февруари', 'Март', 'Април', 'Май', 'Юни',
        'Юли', 'Август', 'Септември', 'Октомври', 'Ноември', 'Декември'],
    onChange: (value) => {
        console.log('Selected datetime:', value);
        document.getElementById('datetime-value').textContent = value || 'None';
    }
});

// DateTime Calendar (12-hour format)
const datetimeCalendar12 = new LumiCalendar({
    target: '#calendar-datetime-12',
    enableDateTime: true,
    hourFormat: '12', // Use 12-hour format
    onChange: (value) => {
        console.log('Selected datetime:', value);
        document.getElementById('datetime-12-value').textContent = value || 'None';
    }
});

// Get current value
const currentValue = dateCalendar.getValue();

// Set a value programmatically
dateCalendar.setValue('2024-12-25');
datetimeCalendar.setValue('2024-12-25 14:30');
datetimeCalendar12.setValue('2024-12-25 14:30');
